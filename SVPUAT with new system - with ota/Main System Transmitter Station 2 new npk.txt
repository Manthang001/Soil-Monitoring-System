// ================= EVALUATOR FUNCTIONS =================
// ---- PCHIP evaluators (specific for EC and pH) ----
float evalPCHIP_EC(float x) {
  const float* breaks = EC_BREAKS;
  const float (*coeffs)[15] = EC_COEFFS;
  if (x <= breaks[0]) {
    float dx = x - breaks[0];
    return coeffs[3][0] + dx * (coeffs[2][0] + dx * (coeffs[1][0] + dx * coeffs[0][0]));
  }
  if (x >= breaks[EC_SEGMENTS]) {
    float dx = x - breaks[EC_SEGMENTS];
    return coeffs[3][EC_SEGMENTS-1] + dx * (coeffs[2][EC_SEGMENTS-1] + dx * (coeffs[1][EC_SEGMENTS-1] + dx * coeffs[0][EC_SEGMENTS-1]));
  }
  for (int i = 0; i < EC_SEGMENTS; i++) {
    if (x >= breaks[i] && x <= breaks[i+1]) {
      float dx = x - breaks[i];
      return coeffs[3][i] + dx * (coeffs[2][i] + dx * (coeffs[1][i] + dx * coeffs[0][i]));
    }
  }
  return 0;
}

float evalPCHIP_pH(float x) {
  const float* breaks = PH_BREAKS;
  const float (*coeffs)[8] = PH_COEFFS;
  if (x <= breaks[0]) {
    float dx = x - breaks[0];
    return coeffs[3][0] + dx * (coeffs[2][0] + dx * (coeffs[1][0] + dx * coeffs[0][0]));
  }
  if (x >= breaks[PH_SEGMENTS]) {
    float dx = x - breaks[PH_SEGMENTS];
    return coeffs[3][PH_SEGMENTS-1] + dx * (coeffs[2][PH_SEGMENTS-1] + dx * (coeffs[1][PH_SEGMENTS-1] + dx * coeffs[0][PH_SEGMENTS-1]));
  }
  for (int i = 0; i < PH_SEGMENTS; i++) {
    if (x >= breaks[i] && x <= breaks[i+1]) {
      float dx = x - breaks[i];
      return coeffs[3][i] + dx * (coeffs[2][i] + dx * (coeffs[1][i] + dx * coeffs[0][i]));
    }
  }
  return 0;
}

// ---- Generic B-spline evaluator ----
float evalBSpline(float x, const float* knots, const float* coeffs, int n_coeffs, int degree) {
  int k = degree;
  while (k < n_coeffs && x > knots[k+1]) k++;
  if (k >= n_coeffs) k = n_coeffs - 1;

  float N[degree+1];
  for (int i = 0; i <= degree; i++) {
    N[i] = (x >= knots[k-i] && x < knots[k-i+1]) ? 1.0f : 0.0f;
  }
  for (int d = 1; d <= degree; d++) {
    for (int i = 0; i <= degree - d; i++) {
      int left = k - i;
      int right = k - i + d;
      float alpha = 0.0f, beta = 0.0f;
      float denom1 = knots[right] - knots[left];
      float denom2 = knots[right+1] - knots[left+1];
      if (denom1 != 0) alpha = (x - knots[left]) / denom1;
      if (denom2 != 0) beta  = (knots[right+1] - x) / denom2;
      N[i] = alpha * N[i] + beta * N[i+1];
    }
  }
  float val = 0.0f;
  for (int i = 0; i <= degree; i++) {
    int idx = k - i;
    if (idx >= 0 && idx < n_coeffs) val += coeffs[idx] * N[i];
  }
  return val;
}

// ================= STATE VARIABLES =================
bool loraReady = false, sdReady = false, localReadSuccess = false, rtcReady = false;
float calMo=0, calTp=0, calPh=0, calEC=0, calN=0, calP=0, calK=0;
float rawMo_tx=0, rawTp_tx=0, rawPh_tx=0, rawEC_tx=0, rawN_tx=0, rawP_tx=0, rawK_tx=0;
int txSuccessCount = 0;
int txFailCount = 0;

// ================= RS485 HELPERS =================
void preTransmission()  { digitalWrite(RS485_DE, HIGH); }
void postTransmission() { digitalWrite(RS485_DE, LOW); }

int readRegister(uint16_t reg) {
  if (node.readHoldingRegisters(reg, 1) == node.ku8MBSuccess)
    return node.getResponseBuffer(0);
  return -1;
}

// ================= SD LOGGER =================
void logToSD(String type, String data) {
  if (!sdReady) return;
  File f = SD.open("/datalog.csv", FILE_APPEND);
  if (f) {
    String timeStr = "0000/00/00 00:00:00";
    if (rtcReady) {
      DateTime now = rtc.now();
      char buf[22];
      sprintf(buf, "%04d/%02d/%02d %02d:%02d:%02d",
              now.year(), now.month(), now.day(),
              now.hour(), now.minute(), now.second());
      timeStr = String(buf);
    }
    f.printf("%s,%s,%s\n", timeStr.c_str(), type.c_str(), data.c_str());
    f.close();
  }
}

// ================= DISPLAY =================
void updateDisplay() {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SH110X_WHITE);

  display.setCursor(0, 0);
  if (rtcReady) {
    DateTime now = rtc.now();
    display.printf("%02d:%02d S-ID:%d LR:%d", now.hour(), now.minute(), STATION_ID, loraReady);
  } else {
    display.printf("S-ID:%d LR:%d SD:%d", STATION_ID, loraReady, sdReady);
  }
  display.drawLine(0, 9, 127, 9, SH110X_WHITE);

  if (localReadSuccess) {
    display.setCursor(0, 13); display.printf("T:%.1f M:%.1f%%", calTp, calMo);
    display.setCursor(0, 23); display.printf("pH:%.2f EC:%.2f", calPh, calEC);
    display.setCursor(0, 33); display.printf("N:%.0f P:%.0f K:%.0f", calN, calP, calK);
  } else {
    display.setCursor(0, 23); display.print("Sensor Read Failed!");
  }
  display.drawLine(0, 43, 127, 43, SH110X_WHITE);

  display.setCursor(0, 47);
  display.printf("TX OK: %d", txSuccessCount);
  display.setCursor(0, 56);
  display.printf("TX FAIL: %d", txFailCount);
  display.display();
}

// ================= SETUP =================
void setup() {
  Serial.begin(115200);
  Serial.println("\n\n========== SVPUAT TX NODE BOOT ==========");

  pinMode(LED_PIN, OUTPUT);
  pinMode(RS485_DE, OUTPUT); digitalWrite(RS485_DE, LOW);
  pinMode(LORA_NSS, OUTPUT); digitalWrite(LORA_NSS, HIGH);
  pinMode(SD_CS, OUTPUT);    digitalWrite(SD_CS, HIGH);

  Wire.begin(I2C_SDA, I2C_SCL);
  display.begin(0x3C, true);
  display.clearDisplay();
  display.setCursor(0, 10);
  display.print("SVPUAT TX Booting...");
  display.display();

  if (rtc.begin()) {
    rtcReady = true;
    if (rtc.lostPower()) rtc.adjust(DateTime(F(__DATE__), F(__TIME__)));
  }

  rs485Serial.begin(4800, SERIAL_8N1, RS485_RX, RS485_TX);   // 4800 baud rate
  node.begin(1, rs485Serial);
  node.preTransmission(preTransmission);
  node.postTransmission(postTransmission);
  Serial.println("[RS485] Ready at 4800 baud");

  sdSPI.begin(SD_SCK, SD_MISO, SD_MOSI);
  if (SD.begin(SD_CS, sdSPI, 4000000)) {
    sdReady = true;
    if (!SD.exists("/datalog.csv")) {
      File f = SD.open("/datalog.csv", FILE_WRITE);
      if (f) { f.println("timestamp,type,data"); f.close(); }
    }
  }

  loraSPI.begin(LORA_SCK, LORA_MISO, LORA_MOSI);
  if (radio.begin(868.0, 125.0, 9, 7, 0x34, 22) == RADIOLIB_ERR_NONE) {
    loraReady = true;
    Serial.println("[LORA] ✅ Ready at 868.0 MHz (SVPUAT NETWORK)");
  } else {
    Serial.println("[LORA] ❌ Init failed!");
  }

  Serial.println("\n========== SVPUAT TX READY ==========\n");
  updateDisplay();
}

// ================= LOOP =================
void loop() {
  // ── 1. Sensor Read (Har 10 Seconds) ───────────────────────────────────
  if (millis() - lastLocalRead >= 10000) {
    lastLocalRead = millis();

    int mMo = readRegister(0x0000); 
    int mT  = readRegister(0x0001); 
    int mEC = readRegister(0x0002); 
    int mPH = readRegister(0x0003);
    int mN  = readRegister(0x0004); 
    int mP  = readRegister(0x0005); 
    int mK  = readRegister(0x0006);

    if (mMo >= 0 && mT >= 0 && mEC >= 0 && mPH >= 0 && mN >= 0 && mP >= 0 && mK >= 0) {
      localReadSuccess = true;

      float rawMo = mMo / 10.0f;
      float rawTp = mT  / 10.0f;
      float rawPh = mPH / 10.0f;
      float rawEC = (float)mEC / 100.0f;
      float rawN  = (float)mN;
      float rawP  = (float)mP;
      float rawK  = (float)mK;

      // Save raw values for LoRa transmission
      rawMo_tx = rawMo;
      rawTp_tx = rawTp;
      rawPh_tx = rawPh;
      rawEC_tx = rawEC;
      rawN_tx  = rawN;
      rawP_tx  = rawP;
      rawK_tx  = rawK;

      // ── Calibration using SPMS v9.3 Models ─────────────────────────────
      calTp = rawTp;   // temperature has no calibration
      if (rawMo < 3.0f) {
        calMo = 0; calPh = 0; calEC = 0;
        calN = 0; calP = 0; calK = 0;
      } else {
        calMo = evalBSpline(rawMo, MOISTURE_KNOTS, MOISTURE_BCOEFFS, sizeof(MOISTURE_BCOEFFS)/sizeof(float), MOISTURE_DEGREE);
        calPh = evalPCHIP_pH(rawPh);
        calEC = evalPCHIP_EC(rawEC);
        calN  = evalBSpline(rawN, N_KNOTS, N_BCOEFFS, sizeof(N_BCOEFFS)/sizeof(float), N_DEGREE);
        calP  = evalBSpline(rawP, P_KNOTS, P_BCOEFFS, sizeof(P_BCOEFFS)/sizeof(float), P_DEGREE);
        calK  = evalBSpline(K_KNOTS, K_BCOEFFS, sizeof(K_BCOEFFS)/sizeof(float), K_DEGREE); // Fixed array ref to K
      }
      if (calP < 0) calP = 0;

      // Log calibrated data to SD
      String sensorData = String(calTp,1) + "," + String(calMo,1) + "," +
                          String(calPh,2) + "," + String(calEC,3) + "," +
                          String(calN,0)  + "," + String(calP,0)  + "," + String(calK,0);
      logToSD("SENSOR_READ", sensorData);

    } else {
      localReadSuccess = false;
      logToSD("SENSOR_ERROR", "Modbus_fail");
      Serial.println("❌ Sensor read failed");
    }
    updateDisplay();
  }

  // ── 2. LoRa Transmission (Har 30 Seconds) ─────────────────────────────
  if (millis() - lastTxTime >= TX_INTERVAL) {
    lastTxTime = millis();

    if (loraReady && localReadSuccess) {
      String txPayload = "SVPUAT_" + String(STATION_ID) + "," +
                         String(rawTp_tx,1) + "," +
                         String(rawMo_tx,1) + "," +
                         String(rawPh_tx,2) + "," +
                         String(rawEC_tx,3) + "," +
                         String(rawN_tx,0)  + "," +
                         String(rawP_tx,0)  + "," +
                         String(rawK_tx,0);

      Serial.println("\n📡 [LORA TX] Sending RAW Data: " + txPayload);
      digitalWrite(LED_PIN, HIGH);
      int state = radio.transmit(txPayload);
      digitalWrite(LED_PIN, LOW);

      if (state == RADIOLIB_ERR_NONE) {
        Serial.println("[LORA] ✅ Transmission successful!");
        txSuccessCount++;
        logToSD("LORA_TX_OK", txPayload);
      } else {
        Serial.printf("[LORA] ❌ Transmission failed, code: %d\n", state);
        txFailCount++;
        logToSD("LORA_TX_FAIL", "Error_" + String(state));
      }
      updateDisplay();
    } else if (!localReadSuccess) {
      Serial.println("\n⚠️ [LORA TX] Skipped - Sensor Error");
    }
  }
}
